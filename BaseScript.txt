var DIS_MELEE = 5
var DIS_FOE = 22

loadout 1

?loc = mine
^&loc.begin
  brew stone

?foe.count = 0
^|foe.distance > DIS_FOE
  >h0,-2,#yellow,Hurry up!
  equipR star_stone
  equipL triskelion

?foe.count ! 0
^&foe.distance <= 20
  ?item.GetCooldown("blade") <= 0
  ^&foe.count >= 7
    equip blade
    activate R

  ?item.GetCooldown("mask") <= 0
    equip mask
    activate R

  ?item.GetCooldown("bardiche") <= 0
  ^&foe.distance <= DIS_MELEE
    equip bardiche
    activate R

  ?loc = bronze_guardian
  ^&foe = "Bronze Guardian"
    equip Vulcan
    ?foe.state = 2
    ^&foe.time = 1
      activate Potion
    ?foe.state = 32
    ^&foe.time = 30
      equipR Mind Stone
