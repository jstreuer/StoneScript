const = import Constants

loadout 1

?foe.count = 0
^|foe.distance > const.DIS_FOE
  > 0,0,Hurry up!
  equipR star_stone
  equipL triskelion

?foe.count ! 0
^&foe.distance <= 20
  ?item.GetCooldown("blade") <= 0
  ^&foe.count >= 7
    equip blade
    activate R

  ?item.GetCooldown("mask") <= 0
    equip mask
    activate R

  ?item.GetCooldown("bardiche") <= 0
  ^&foe.distance <= const.DIS_MELEE
    equip bardiche
    activate R
